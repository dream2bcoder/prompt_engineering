CREATE TABLE `audit_map` (
  `id` int NOT NULL AUTO_INCREMENT,
  `table_name` varchar(100) DEFAULT NULL,
  `table_pk` varchar(100) DEFAULT NULL,
  `field_name` varchar(100) DEFAULT NULL,
  `field` varchar(100) DEFAULT NULL,
  `screen` varchar(100) DEFAULT NULL,
  `section` varchar(100) DEFAULT NULL,
  PRIMARY KEY (`id`),
  UNIQUE KEY `idx_tbl_field` (`table_name`,`field_name`)
) ENGINE=InnoDB AUTO_INCREMENT=16 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci;

CREATE TABLE `lookup_reference` (
  `id` int NOT NULL AUTO_INCREMENT,
  `category` varchar(50) NOT NULL,
  `subcategory` varchar(50) DEFAULT NULL,
  `lookup_code` varchar(50) DEFAULT NULL,
  `lookup_value` varchar(255) DEFAULT NULL,
  `is_active` tinyint NOT NULL DEFAULT '1',
  `created_dttm` datetime DEFAULT CURRENT_TIMESTAMP,
  `modified_dttm` datetime DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  PRIMARY KEY (`id`),
  KEY `category` (`category`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci;

CREATE TABLE `lookup_reference_detail` (
  `id` int NOT NULL AUTO_INCREMENT,
  `reference_id` int NOT NULL,
  `group_id` int DEFAULT NULL,
  `group_type` varchar(50) DEFAULT NULL,
  `key` varchar(100) DEFAULT NULL,
  `value` text,
  `is_active` tinyint NOT NULL DEFAULT '1',
  `created_dttm` datetime DEFAULT CURRENT_TIMESTAMP,
  `modified_dttm` datetime DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  PRIMARY KEY (`id`),
  KEY `reference_id` (`reference_id`),
  KEY `group_type` (`group_type`),
  KEY `group_id` (`group_id`),
  KEY `reference_id_2` (`reference_id`,`group_id`,`group_type`),
  CONSTRAINT `lookup_reference_detail_ibfk_1` FOREIGN KEY (`reference_id`) REFERENCES `lookup_reference` (`id`) ON DELETE CASCADE ON UPDATE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci;

CREATE TABLE `audit_lookup_reference` (
  `auditAction` enum('INSERT','UPDATE','DELETE') DEFAULT NULL,
  `auditTimestamp` datetime DEFAULT CURRENT_TIMESTAMP,
  `auditId` int NOT NULL AUTO_INCREMENT,
  `id` int NOT NULL,
  `category` varchar(50) NOT NULL,
  `subcategory` varchar(50) DEFAULT NULL,
  `lookup_code` varchar(50) DEFAULT NULL,
  `lookup_value` varchar(255) DEFAULT NULL,
  `is_active` tinyint NOT NULL DEFAULT '1',
  `created_dttm` datetime DEFAULT CURRENT_TIMESTAMP,
  `modified_dttm` datetime DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY (`auditId`),
  KEY `auditTimestamp` (`auditTimestamp`),
  KEY `auditAction` (`auditAction`),
  KEY `id` (`id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci;
